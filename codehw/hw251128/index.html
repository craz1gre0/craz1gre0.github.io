import tkinter as tk
import csv
import random

root= tk.Tk()
root.title("抽籤小工具") # 視窗標題
root.geometry("550x350") # 視窗大小
root.resizable(False, False) # 固定視窗大小
root.configure(bg="#1e1e2f")  # 深色背景

participants= []
try: # 讀取csv檔案
    with open("studets.csv", "r", encoding="utf-8") as f:
        reader= csv.DictReader(f)
        for row in reader:
            student= row["姓名"].strip()
            if student:
                participants.append(student)
except:
    participants= ["名單讀取錯誤"]

rolling= False  # 是否正在抽籤

title_label= tk.Label( # 標題
    root,
    text="抽籤小工具",
    font=("Arial", 24, "bold"),
    fg="#ffd700", # 金色文字
    bg="#1e1e2f" # 深色背景
)
title_label.pack(pady=15) # 上下間距

name_display= tk.Label( # 顯示抽中的名字
    root,
    text="按下「開始抽籤」",
    font=("Arial", 32, "bold"),
    fg="#ffffff", # 白色文字
    bg="#2e2e3e", # 深色背景
    width=15,
    height=2,
    relief="ridge", # 邊框樣式
    bd=4 # 邊框寬度
)
name_display.pack(pady=25) # 上下間距

def start_lottery(): # 開始抽籤
    global rolling
    if rolling: # 如果正在抽籤，就忽略點擊
        return

    rolling= True # 設定為正在抽籤
    start_btn.config(state="disabled", text="抽籤中...") # 按鈕變為不可點擊，顯示抽籤中...
    animate_names(0) # 開始動畫

def animate_names(step): # 抽籤動畫
    global rolling
    if step < 25:  # 抽籤動畫持續25次
        name_display.config(text=random.choice(participants)) # 顯示隨機名字
        root.after(70, lambda: animate_names(step + 1)) # 70毫秒後繼續下一次動畫
    else:
        
        available= [n for n in participants if n != "郭冠廷"] # 避免抽到帥哥
        if not available:
            available= participants # 如果名單中只有帥哥，就只能抽他

        final_name= random.choice(available) # 最終抽中的名字
        name_display.config(text=final_name) # 顯示最終名字
        start_btn.config(state="normal", text="開始抽籤") # 按鈕恢復可點擊，顯示開始抽籤
        rolling= False # 抽籤結束

start_btn= tk.Button( # 開始抽籤按鈕
    root,
    text="開始抽籤",
    font=("Arial", 18, "bold"),
    bg="#4caf50", # 按鈕背景色
    fg="#ffffff", # 文字顏色
    activebackground="#45a049", # 按鈕按下時背景色
    activeforeground="#ffffff", # 按鈕按下時文字顏色
    relief="raised", # 按鈕邊框樣式
    bd=3, # 按鈕邊框寬度
    command=start_lottery # 點擊按鈕時執行的函式
)
start_btn.pack(pady=15) # 上下間距

root.mainloop()
