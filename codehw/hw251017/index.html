class Product:
  def __init__(self,name,price,seller): #題目要求屬性(here)
    self.name= name
    self.price= price
    self.seller= seller
    self.is_sold= False 
    #要預設false所以上面不用(here)
    #沒意義，然後也是看他mouse = Product("羅技滑鼠", 300, "小明")
  
  def show_info(self):
        status= "已售出" if self.is_sold else "販售中" #A if 條件 else B
        print(f"商品：{self.name} | 價格：{self.price}元 | 賣家：{self.seller} | 狀態：{status}")

  def mark_as_sold(self):
    self.is_sold= True
    
    
class User:
  def __init__(self,username):
    self.username= username
    self.my_products= []  #設list
    
  def add_product(self,product):
    self.my_products.append(product)
    
  def buy_product(self,product):
    product.mark_as_sold() 
    #因為book = Product("微積分課本", 200, "小明")，所以book被設為一個 Product 物件
    #然後Python 會找到 book 所屬的類別(Product)，找到裡面的mark_as_sold
    
    print("購買成功！")
    
  def show_my_products(self):
    for 商品 in self.my_products:
      商品.show_info()
      #跟上面依樣 商品 是一個 Product 物件，可以直接呼叫
      
      
      
# 建立使用者
student1= User("小明")
student2= User("小華")

# 小明刊登商品
book= Product("微積分課本", 200, "小明")

# 顯示小明的商品
print("=== 小明的商品 ===")

# 小華購買商品
print("\n=== 小華購買微積分課本 ===")
